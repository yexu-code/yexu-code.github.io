1.3 上传文件

在SFTPUtil.java中新建一个上传文件的方法upload()，利用ChannelSftp类中的put方法上传文件。

public void upload(String basePath, String directory, String sftpFileName, InputStream input) throws SftpException{
    try {
        sftp.cd(basePath);
        sftp.cd(directory);
    } catch (SftpException e) {
        //目录不存在，则创建文件夹
        String [] dirs = directory.split("/");
        String tempPath = basePath;
        for(String dir:dirs){
            if(null == dir || "".equals(dir)) continue;
            tempPath += "/" + dir;
            try{
                sftp.cd(tempPath);
            }catch(SftpException ex){
                sftp.mkdir(tempPath);
                sftp.cd(tempPath);
            }
        }
    }
    sftp.put(input, sftpFileName);  //上传文件
    System.out.println("上传文件至 " + basePath + directory + "/" + sftpFileName + " 成功!");
}

现有一个文件AUDIT_ECEMOP_S_IAAS_9999992_RP-25_20211223_01_045952_05-01，包含1万条记录。目标是每2000条记录拆分成一个子文件，所以共可以分成5个包含2000条数据的子文件。

编写一个拆分文件的方法splitFile()，能够按照指定记录数拆分文件。

public void splitFile(int recordsNum, String sourceDir, String targetDir) throws IOException {
        Date date = new Date();
        // 源文件
        File sourceFile = new File(sourceDir);
        FileInputStream fis = new FileInputStream(sourceFile);
        BufferedReader br = new BufferedReader(new InputStreamReader(fis));

        // 计算源文件总记录数
        int lineCount = 0;
        FileReader fr = new FileReader(sourceFile);
        LineNumberReader lnr = new LineNumberReader(fr);
        lnr.skip(Long.MAX_VALUE);
        lineCount = lnr.getLineNumber();
        lnr.close();

        // 需生成的文件数
        int fileNum = lineCount / recordsNum + (lineCount % recordsNum == 0 ? 0 : 1);
        int recordsCount = 0; // 已读取的记录数

        String line = null;
        for (int fileCount = 1; fileCount <= fileNum; fileCount++){
            // 目标文件
            File targetFile = new File(targetDir + "_" + sdf.format(date) + "_" + lineCount + "_" + fileNum + "-" + fileCount);
            if (!targetFile.exists()){
                targetFile.createNewFile();
            }
            FileOutputStream fos = new FileOutputStream(targetFile);
            BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(fos));

            while ((line = br.readLine()) != null) {
                bw.write(line);
                recordsCount++;
                if (recordsCount == recordsNum) {
                    recordsCount = 0;
                    bw.flush();
                    bw.close();
                    break;
                } else if (recordsCount < recordsNum){
                    bw.newLine(); 
                }
            }
            bw.close(); 
            System.out.println("第" + fileCount + "个文件已生成.");
        }
        br.close();
        System.out.println("文件拆分结束!");
    }

2.2 压缩文件

编写一个压缩文件的方法compressFile()，将文件压缩为gzip格式。

public void compressFile(String sourceDir, String targetDir) throws IOException {
    // 源文件
    File sourceFile = new File(sourceDir);
    FileInputStream fis = new FileInputStream(sourceFile);
    // 目标文件
    File targetFile = new File(targetDir);
    if (!targetFile.exists()){
        targetFile.createNewFile();
    }
    FileOutputStream fos = new FileOutputStream(targetFile);
    GZIPOutputStream gos = new GZIPOutputStream(fos);
    byte[] buffer = new byte[1024];
    while(fis.read(buffer) > 0){
        gos.write(buffer);
    }
    fis.close();
    gos.finish();
    gos.close();
    System.out.println("文件压缩成功!");
}

3 数据稽核

3.1 构造数据

首先新建一个数据库，在数据库中新建两张表。

CREATE TABLE `console_os_biz_vm_host` (
`ID` varchar(50) NOT NULL COMMENT '唯一标示，资源编码',
`NAME` varchar(100) DEFAULT NULL COMMENT '虚拟机名称',
`IS_DELETE` int(2) DEFAULT '0' COMMENT '是否删除 1删除0未删除',
PRIMARY KEY (`ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='控制台虚拟机表';

CREATE TABLE `openstack_os_biz_vm_host` (
`ID` varchar(50) NOT NULL COMMENT '唯一标示，资源编码',
`NAME` varchar(100) DEFAULT NULL COMMENT '虚拟机名称',
`IS_DELETE` int(2) DEFAULT '0' COMMENT '是否删除 1删除0未删除',
PRIMARY KEY (`ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='底层虚拟机表';

数据稽核的目的是找出控制台已删除，但是底层未删除的资源（控制台和底层资源不一致）。具体来说：

  • 控制台已删除: 控制台虚拟机表中对应记录IS_DELETE字段为1
  • 底层未删除: 底层虚拟机表中对应记录IS_DELETE字段为0

3.2 数据稽核

如前所述，目的是稽核出控制台已删除，但是底层未删除的资源。也即，在控制台虚拟机表中IS_DELETE字段为1，但在底层虚拟机表中IS_DELETE字段为0的记录。于是可以写出如下的SQL语句。

SELECT o.ID, o.NAME, o.IS_DELETE
FROM console_os_biz_vm_host c, openstack_os_biz_vm_host o
WHERE c.ID = o.ID AND c.IS_DELETE = 1 AND o.IS_DELETE = 0;

在UserService类中编写数据稽核的接口方法queryInconsistentData，用于表数据稽核，并将稽核出的数据存入List集合中。

3.3 将数据写入到文件

由于上一步已经将稽核出的数据记录存入了List集合，所以需要把List集合中的数据写入到文件中。在FileUtil类中编写一个方法list2File，能够将List集合中的数据写入到目标文件。

public void list2File(List<?> list, String targetDir){
    // 目标文件
    File targetFile = new File(targetDir);
    if (!targetFile.exists()){
        try {
            targetFile.createNewFile();
        } catch (IOException e) {
            System.out.println("文件创建失败!" + e.getMessage());
        }
    }
    try {
        Writer writer = new FileWriter(targetFile);
        BufferedWriter bw = new BufferedWriter(writer);
        for (OpenstackOsBizVmHost os : (List<OpenstackOsBizVmHost>)list){
            bw.write(os.getId() + "|" + os.getName() + "|" + os.getIsDelete());
            bw.newLine();
        }
        bw.close();
    } catch (IOException e) {
        e.printStackTrace();
    }
    System.out.println("数据写入完成!");
}

public class Test04 {
    /**
     * 总体测试程序
     * Author: LiLei  Data: 2022/7/26
     */
    private FileUtils fileUtils = new FileUtils();

    private String username = "******";
    private String password = "******";
    private String host = "******";
    private int port = **;

    private String homePath = "D:\\data\\";

    @Test
    public void dataAuditing() throws IOException {

        String config = "applicationContext.xml";
        ApplicationContext ctx = new ClassPathXmlApplicationContext(config);
        UserService userService = (UserService) ctx.getBean("userService");

        // 1.数据稽核,数据存入List集合
        List<OpenstackOsBizVmHost> listData = userService.queryInconsistentData();
        System.out.println("共稽核出 " + + listData.size() + " 条不一致数据.");

        // 2.将稽核出的List集合中的数据写入文件
        String targetDir = homePath + "test_inconsistentData";
        fileUtils.list2File(listData, targetDir);

        // 3.拆分文件
        int lineCount = 0; // 源文件总记录数
        FileReader fr = new FileReader(targetDir);
        LineNumberReader lnr = new LineNumberReader(fr);
        lnr.skip(Long.MAX_VALUE);
        lineCount = lnr.getLineNumber();
        lnr.close();

        int recordsNum = 2000; // 拆分后子文件的记录数
        int fileNum = lineCount / recordsNum + (lineCount % recordsNum == 0 ? 0 : 1); // 需生成的文件数

        String sourceDir = homePath + "test_inconsistentData";
        targetDir = homePath + "test_inconsistentDataSplit";
        fileUtils.splitFile(recordsNum, sourceDir, targetDir);

        // 4.压缩文件
        for (int i = 1; i <= fileNum; i++){
            sourceDir = homePath + "test_inconsistentDataSplit_20220726_9200_5-" + i;
            targetDir = homePath + "compress\\test_inconsistentDataSplit_20220726_9200_5-" + i +".gz";
            fileUtils.compressFile(sourceDir, targetDir);
        }

        // 5.将压缩后的文件上传至SFTP服务器
        SFTPUtil sftp = new SFTPUtil(username, password, host, port);
        sftp.login();

        for(int i = 1; i <= fileNum; i++){
            String fileName = "test_inconsistentDataSplit_20220726_9200_5-";
            File file = new File(homePath + "compress\\" + fileName + i +".gz");
            InputStream is = null;
            try {
                is = new FileInputStream(file);
            } catch (FileNotFoundException e) {
                e.printStackTrace();
            }

            String sftpFileName = fileName + i + ".gz";
            try {
                sftp.upload("/","/upload", sftpFileName, is);
                is.close();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }

        sftp.logout();
    }

}
